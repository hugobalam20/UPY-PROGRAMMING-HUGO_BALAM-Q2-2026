INICIO

    DEFINIR users COMO DICCIONARIO
    DEFINIR subjects COMO TUPLA
    subjects <- ("Discrete Mathematics", "Programming", "English II", "Differential Calculus", "Probability and Statistics", "Computer and Server Architecture", "Socio-Emotional Skills and Conflict Management")
    
    DEFINIR notes COMO DICCIONARIO

    DEFINIR program_active COMO BOOLEANO
    program_active <- VERDADERO

    MIENTRAS program_active = VERDADERO HACER
        
        DEFINIR current_user, current_role, current_name COMO CADENA
        DEFINIR authenticated COMO BOOLEANO
        authenticated <- FALSO

        ESCRIBIR "--- SCHOOL MANAGEMENT SYSTEM - LOGIN ---"
        ESCRIBIR "(Type 'shutdown' in username to turn off the program)"

        MIENTRAS authenticated = FALSO HACER
            LEER username_input
            
            SI username_input = "shutdown" ENTONCES
                ESCRIBIR "Shutting down the system. Goodbye!"
                authenticated <- VERDADERO
                program_active <- FALSO
            SINO
                LEER password_input
                
                SI existe username_input EN users Y users[username_input]["password"] = password_input ENTONCES
                    current_user <- username_input
                    current_role <- users[username_input]["rol"]
                    current_name <- users[username_input]["name"]
                    
                    ESCRIBIR "Welcome, ", current_name, " (", current_role, ")"
                    authenticated <- VERDADERO
                SINO
                    ESCRIBIR "Incorrect username or password. Try again."
                FIN SI
            FIN SI
        FIN MIENTRAS

        SI program_active = FALSO ENTONCES
            ROMPER BUCLE
        FIN SI

        SI current_role = "student" ENTONCES
            ESCRIBIR "--- REPORT CARD: ", current_name, " ---"
            
            DEFINIR aprobadas COMO CONJUNTO
            DEFINIR all_subjects COMO CONJUNTO
            DEFINIR pendientes COMO CONJUNTO
            
            PARA CADA subject EN subjects HACER
                DEFINIR grade COMO REAL
                grade <- notes[current_user][subject]
                ESCRIBIR "  * ", subject, ": ", grade
                
                SI grade >= 7.0 ENTONCES
                    AGREGAR subject A aprobadas
                FIN SI
            FIN PARA
            
            all_subjects <- CONVERTIR subjects A CONJUNTO
            pendientes <- all_subjects DIFERENCIA aprobadas
            
            ESCRIBIR "Passed subjects (aprobadas): ", aprobadas
            ESCRIBIR "Pending subjects (pendientes): ", pendientes
            
            ESCRIBIR "Press Enter to log out and return to Login..."
            LEER pausa

        SINO SI current_role = "professor" ENTONCES
            ESCRIBIR "--- List of Registered Students ---"
            PARA CADA usr EN users HACER
                SI users[usr]["rol"] = "student" ENTONCES
                    ESCRIBIR "  - Username: ", usr, " | Name: ", users[usr]["name"]
                FIN SI
            FIN PARA
            
            DEFINIR menu_professor_active COMO BOOLEANO
            menu_professor_active <- VERDADERO
            
            MIENTRAS menu_professor_active = VERDADERO HACER
                ESCRIBIR "--- Grade Modification Menu ---"
                ESCRIBIR "(Type 'logout' in student username to close professor session)"
                LEER student_selec
                
                SI student_selec = "logout" ENTONCES
                    ESCRIBIR "Logging out from professor session..."
                    menu_professor_active <- FALSO
                    
                SINO SI existe student_selec EN users Y users[student_selec]["rol"] = "student" ENTONCES
                    ESCRIBIR "Available subjects to grade:"
                    PARA CADA s EN subjects HACER
                        ESCRIBIR "  * ", s
                    FIN PARA
                    
                    LEER subject_selec
                    
                    SI existe subject_selec EN subjects ENTONCES
                        DEFINIR grade_confirmed COMO BOOLEANO
                        grade_confirmed <- FALSO
                        
                        MIENTRAS grade_confirmed = FALSO HACER
                            DEFINIR new_grade COMO REAL
                            LEER new_grade
                            
                            ESCRIBIR "Are you sure you want to change ", subject_selec, " grade to ", new_grade, "? (s/n):"
                            LEER confirmation
                            
                            SI confirmation = "s" ENTONCES
                                notes[student_selec][subject_selec] <- new_grade
                                ESCRIBIR "Grade updated successfully!"
                                
                                ESCRIBIR "--- Updated Report Card for ", users[student_selec]["name"], " ---"
                                PARA CADA s EN subjects HACER
                                    ESCRIBIR "  - ", s, ": ", notes[student_selec][s]
                                FIN PARA
                                
                                grade_confirmed <- VERDADERO
                            SINO
                                ESCRIBIR "[Correction] Re-entering the grade for the same student and subject..."
                            FIN SI
                        FIN MIENTRAS
                    SINO
                        ESCRIBIR "[Error] The entered subject is not valid. Try again."
                    FIN SI
                SINO
                    ESCRIBIR "[Error] User does not exist or is not a student. Try again."
                FIN SI
            FIN MIENTRAS

        SINO SI current_role = "coordinator" ENTONCES
            ESCRIBIR "========================================="
            ESCRIBIR "      GLOBAL REPORT (READ-ONLY)          "
            ESCRIBIR "========================================="
            
            ESCRIBIR "1. LIST OF PROFESSORS:"
            PARA CADA usr EN users HACER
                SI users[usr]["rol"] = "professor" ENTONCES
                    ESCRIBIR "   * ", users[usr]["name"]
                FIN SI
            FIN PARA
            
            ESCRIBIR "2. LIST OF SUBJECTS:"
            PARA CADA subject EN subjects HACER
                ESCRIBIR "   * ", subject
            FIN PARA
            
            ESCRIBIR "3. STUDENT GRADES REPORT:"
            PARA CADA student EN notes HACER
                ESCRIBIR "   Student: ", users[student]["name"], " (", student, ")"
                PARA CADA s EN subjects HACER
                    ESCRIBIR "      - ", s, ": ", notes[student][s]
                FIN PARA
            FIN PARA
            
            ESCRIBIR "Press Enter to log out and return to Login..."
            LEER pausa
            
        SINO
            ESCRIBIR "[X] Role not recognized in the system."
        FIN SI

    FIN MIENTRAS

FIN
